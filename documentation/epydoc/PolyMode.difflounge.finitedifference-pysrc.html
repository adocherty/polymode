 15          ''' 
 16          Calculate the Finite Difference weights for a FD stencil on a base of points x 
 17          at the locations x0. Calculates m levels of stencils (m=0 corresponding to the identity) 
 18           
 19          Algorithm taken from: 
 20          B. Fornberg, “Generation of Finite Difference Formulas on Arbitrarily Spaced Grids,” 
 21          Mathematics of Computation 51, no. 184 (1988): 699-706. 
 22          ''' 
 23          x=asanyarray(x) 
 24          n = len(x) 
 25   
 26          weights = zeros((m+1,n), dtype=dtype) 
 27          weights[0,0] = 1. 
 28          betaold = 1. 
 29          for i in range(1,n): 
 30                  beta = prod(x[i]-x[0:i]) 
 31                  for k in range(0,min(i,m)+1): 
 32                          weights[k,i] = betaold*(k*weights[k-1,i-1]-(x[i-1]-x0)*weights[k,i-1])/beta 
 33                  betaold=beta 
 34   
 35                  for j in range(0,i): 
 36                          for k in range(min(i,m),-1,-1): 
 37                                  weights[k,j] = ((x[i]-x0)*weights[k,j]-k*weights[k-1,j])/(x[i]-x[j]) 
 38   
 39          # Clear very small entries: 
 40          weights[absolute(weights)<1e-10] = 0 
 41          return weights
